```cpp

#include <bits/stdc++.h>
using namespace std;


int lengthOfLastWord(string s) {
    int lastToF = s.size() - 1;
    int cntAns = 0;

    while(lastToF >= 0 && s[lastToF] == ' ') {
        lastToF--;
    }

    while(lastToF >= 0 && s[lastToF] != ' ') {
        cntAns++;
        lastToF--;
    }
    return cntAns;
}


int main() {

  string str = "Mehul  Thuletiya   ";
  cout << "Length of Last Word: "<< lengthOfLastWord(str);

  return 0;
}
```
