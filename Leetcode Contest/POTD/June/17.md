## 2 pointer

```cpp

class Solution {
public:
    bool judgeSquareSum(int c) {
        long long int left = 0, right = sqrt(c);

        while (left <= right) {
            if (left * left + right * right == c)
                return true;
            else if (left * left + right * right > c)
                right--;
            else
                left++;
        }
        return false;
    }
};

// TC: O(logN + sqrt(c) )
// SC: O(1)

```
